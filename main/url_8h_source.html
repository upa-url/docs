921// Removable URL chars
922
923// chars to trim (C0 control or space: U+0000 to U+001F or U+0020)
924template <typename CharT>
925constexpr bool is_trim_char(CharT ch) noexcept {
926 return util::to_unsigned(ch) <= ' ';
927}
928
929// chars what should be removed from the URL (ASCII tab or newline: U+0009, U+000A, U+000D)
930template <typename CharT>
931constexpr bool is_removable_char(CharT ch) noexcept {
932 return ch == '\r' || ch == '\n' || ch == '\t';
933}
934
935template <typename CharT>
936inline void do_trim(const CharT*& first, const CharT*& last) noexcept {
937 // remove leading C0 controls and space
938 while (first < last && is_trim_char(*first))
939 ++first;
940 // remove trailing C0 controls and space
941 while (first < last && is_trim_char(*(last-1)))
942 --last;
943}
944
945// DoRemoveURLWhitespace
946// https://cs.chromium.org/chromium/src/url/url_canon_etc.cc
947template <typename CharT>
948inline void do_remove_whitespace(const CharT*& first, const CharT*& last, simple_buffer<CharT>& buff) {
949 // Fast verification that there's nothing that needs removal. This is the 99%
950 // case, so we want it to be fast and don't care about impacting the speed
951 // when we do find whitespace.
952 for (auto it = first; it < last; ++it) {
953 if (!is_removable_char(*it))
954 continue;
955 // copy non whitespace chars into the new buffer and return it
956 buff.reserve(last - first);
957 buff.append(first, it);
958 for (; it < last; ++it) {
959 if (!is_removable_char(*it))
960 buff.push_back(*it);
961 }
962 first = buff.data();
963 last = buff.data() + buff.size();
964 break;
965 }
966}

1017constexpr bool is_windows_drive(CharT c1, CharT c2) noexcept {
1018 return is_ascii_alpha(c1) && (c2 == ':' || c2 == '|');
1019}

1021// https://url.spec.whatwg.org/#normalized-windows-drive-letter
1022template <typename CharT>
1023constexpr bool is_normalized_windows_drive(CharT c1, CharT c2) noexcept {

1025}

1027// https://url.spec.whatwg.org/#start-with-a-windows-drive-letter
1028template <typename CharT>
1029inline bool starts_with_windows_drive(const CharT* pointer, const CharT* last) noexcept {
1030 const auto length = last - pointer;
1031#if 1
1032 return
1033 (length == 2 || (length > 2 && detail::is_special_authority_end_char(pointer[2]))) &&
1034 detail::is_windows_drive(pointer[0], pointer[1]);
1035#else
1036 return
1037 length >= 2 &&
1038 detail::is_windows_drive(pointer[0], pointer[1]) &&
1039 (length == 2 || detail::is_special_authority_end_char(pointer[2]));
1040#endif
1041}

1043// Windows drive letter in OS path
1044//
1045// NOTE: Windows OS supports only normalized Windows drive letters.
1046
1047// Check url's pathname has Windows drive, i.e. starts with "/C:/" or is "/C:"
1048// see also: detail::starts_with_windows_drive
1049inline bool pathname_has_windows_os_drive(string_view pathname) noexcept {
1050 return
1051 (pathname.length() == 3 || (pathname.length() > 3 && is_windows_slash(pathname[3]))) &&
1052 is_windows_slash(pathname[0]) &&
1053 is_normalized_windows_drive(pathname[1], pathname[2]);
1054}

1059template <typename CharT>
1060constexpr const CharT* is_windows_os_drive_absolute_path(const CharT* pointer, const CharT* last) noexcept {
1061 return (last - pointer > 2 &&
1062 is_normalized_windows_drive(pointer[0], pointer[1]) &&
1063 is_windows_slash(pointer[2]))
1064 ? pointer + 3 : nullptr;
1065}

1385// Implements "basic URL parser" https://url.spec.whatwg.org/#concept-basic-url-parser
1386// without encoding, url and state override arguments. It resets this url object to
1387// an empty value and then parses the input and modifies this url object.
1388// Returns validation_errc::ok on success, or an error value on parsing failure.
1389template <typename CharT>
1390inline validation_errc url::do_parse(const CharT* first, const CharT* last, const url* base) {
1391 const validation_errc res = [&]() {
1392 detail::url_serializer urls(*this);
1393
1394 // reset URL
1395 urls.new_url();
1396
1397 // is base URL valid?
1398 if (base && !base->is_valid())
1399 return validation_errc::invalid_base;
1400
1401 // remove any leading and trailing C0 control or space:
1402 detail::do_trim(first, last);
1403 //TODO-WARN: validation error if trimmed
1404
1405 return detail::url_parser::url_parse(urls, first, last, base);
1406 }();
1407 if (res == validation_errc::ok) {
1408 set_flag(VALID_FLAG);
1409 parse_search_params();
1410 }
1411 return res;
1412}

1598// Implements "basic URL parser" https://url.spec.whatwg.org/#concept-basic-url-parser
1599// without 1 step. It modifies the URL stored in the urls object.
1600// Returns validation_errc::ok on success, or an error value on parsing failure.
1601template <typename CharT>
1602inline validation_errc url_parser::url_parse(url_serializer& urls, const CharT* first, const CharT* last, const url* base, State state_override)
1603{
1604 using UCharT = typename std::make_unsigned<CharT>::type;
1605
1606 // remove all ASCII tab or newline from URL
1607 simple_buffer<CharT> buff_no_ws;
1608 detail::do_remove_whitespace(first, last, buff_no_ws);
1609 //TODO-WARN: validation error if removed
1610
1611 if (urls.need_save()) {
1612 // reserve size (TODO: But what if `base` is used?)
1613 const auto length = std::distance(first, last);
1614 urls.reserve(length + 32);
1615 }
1616
1617#ifdef UPA_URL_USE_ENCODING
1618 const char* encoding = "UTF-8";
1619 // TODO: If encoding override is given, set encoding to the result of getting an output encoding from encoding override.
1620#endif
1621
1622 auto pointer = first;
1623 State state = state_override ? state_override : scheme_start_state;
1624
1625 // has scheme?
1626 if (state == scheme_start_state) {
1627 if (pointer != last && detail::is_first_scheme_char(*pointer)) {
1628 state = scheme_state; // this appends first char to buffer
1629 } else if (!state_override) {
1630 state = no_scheme_state;
1631 } else {
1632 // 3. Otherwise, return failure.
1633 return validation_errc::scheme_invalid_code_point;
1634 }
1635 }
1636
1637 if (state == scheme_state) {
1638 // Deviation from URL stdandart's [ 2. ... if c is ":", run ... ] to
1639 // [ 2. ... if c is ":", or EOF and state override is given, run ... ]
1640 // This lets protocol setter to pass input without adding ':' to the end.
1641 // Similiar deviation exists in nodejs, see:
1642 // https://github.com/nodejs/node/pull/11917#pullrequestreview-28061847
1643
1644 // first scheme char has been checked in the scheme_start_state, so skip it
1645 const auto end_of_scheme = std::find_if_not(pointer + 1, last, detail::is_scheme_char<CharT>);
1646 const bool is_scheme = end_of_scheme != last
1647 ? *end_of_scheme == ':'
1648 : state_override != not_set_state;
1649

2321template <typename CharT>
2322inline void url_parser::parse_path(url_serializer& urls, const CharT* first, const CharT* last) {
2323 // path state; includes:
2324 // 1. [ (/,\‍) - 1, 2, 3, 4 - [ 1 (if first segment), 2 ] ]
2325 // 2. [ 1 ... 4 ]
2326 static const auto escaped_dot = [](const CharT* const pointer) -> bool {
2327 // "%2e" or "%2E"
2328 return pointer[0] == '%' && pointer[1] == '2' && (pointer[2] | 0x20) == 'e';
2329 };
2330 static const auto double_dot = [](const CharT* const pointer, const std::size_t len) -> bool {
2331 switch (len) {
2332 case 2: // ".."
2333 return pointer[0] == '.' && pointer[1] == '.';
2334 case 4: // ".%2e" or "%2e."
2335 return (pointer[0] == '.' && escaped_dot(pointer + 1)) ||
2336 (escaped_dot(pointer) && pointer[3] == '.');
2337 case 6: // "%2e%2e"
2338 return escaped_dot(pointer) && escaped_dot(pointer + 3);
2339 default:
2340 return false;
2341 }
2342 };
2343 static const auto single_dot = [](const CharT* const pointer, const std::size_t len) -> bool {
2344 switch (len) {
2345 case 1: return pointer[0] == '.';
2346 case 3: return escaped_dot(pointer); // "%2e"
2347 default: return false;
2348 }
2349 };
2350
2351 // parse path's segments
2352 auto pointer = first;
2353 while (true) {
2354 const auto end_of_segment = urls.is_special_scheme()
2355 ? std::find_if(pointer, last, detail::is_slash<CharT>)
2356 : std::find(pointer, last, '/');
2357
2358 // end_of_segment >= pointer
2359 const std::size_t len = end_of_segment - pointer;
2360 const bool is_last = end_of_segment == last;
2361 // TODO-WARN: 1. If url is special and c is "\", validation error.
2362
2363 if (double_dot(pointer, len)) {
2364 urls.shorten_path();
2365 if (is_last) urls.append_empty_path_segment();
2366 } else if (single_dot(pointer, len)) {
2367 if (is_last) urls.append_empty_path_segment();
2368 } else {
2369 if (len == 2 &&
2370 urls.is_file_scheme() &&
2371 urls.is_empty_path() &&
2372 detail::is_windows_drive(pointer[0], pointer[1]))
2373 {
2374 // replace the second code point in buffer with ":"
2375 std::string& str_path = urls.start_path_segment();
2376 str_path += static_cast<char>(pointer[0]);
2377 str_path += ':';
2378 urls.save_path_segment();
2379 //Note: This is a (platform-independent) Windows drive letter quirk.
2380 } else {
2381 std::string& str_path = urls.start_path_segment();
2382 do_path_segment(pointer, end_of_segment, str_path);
2383 urls.save_path_segment();
2384 // end of segment
2385 pointer = end_of_segment;
2386 }
2387 }
2388 // next segment
2389 if (is_last) break;
2390 pointer = end_of_segment + 1; // skip '/' or '\'
2391 }
2392}

2697 // https://url.spec.whatwg.org/#concept-url-serializer
2698 const url::PartType ifirst = [&]() {

2700 // authority, host
2701 if (!src.is_null(url::HOST)) {
2702 if (t1 == url::USERNAME && src.has_credentials())
2703 return url::USERNAME;
2704 return url::HOST;
2705 }
2706 return url::PATH_PREFIX;
2707 }
2708 // t1 == PATH
2709 return t1;
2710 }();
2711
2712 if (!need_save()) return;
2713
2714 // copy flags; they can be used when copying / serializing url parts below
2715 unsigned mask = 0;
2716 for (int ind = t1; ind <= t2; ++ind) {
2717 mask |= url::kPartFlagMask[ind];
2718 }
2719 url_.flags_ = (url_.flags_ & ~mask) | (src.flags_ & mask);
2720
2721 // copy parts & str
2722 if (ifirst <= t2) {
2723 int ilast = t2;
2724 for (; ilast >= ifirst; --ilast) {
2725 if (src.part_end_[ilast])
2726 break;
2727 }
2728 if (ifirst <= ilast) {
2729 // prepare buffer to append data
2730 // IMPORTANT: do before any url_ members modifications!

2732
2733 // last part and url_.path_segment_count_
2734 std::size_t lastp_end = src.part_end_[ilast];
2735 if (pathOpFn && ilast == url::PATH) {

2738 // todo: use std::invoke (c++17)

2740 url_.path_segment_count_ = segment_count;
2741 } else if (ifirst <= url::PATH && url::PATH <= ilast) {
2742 url_.path_segment_count_ = src.path_segment_count_;

3019template <typename CharT>
3020inline const CharT* is_unc_path(const CharT* first, const CharT* last)
3021{
3022 // https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-dfsc/149a3039-98ce-491a-9268-2f5ddef08192
3023 std::size_t path_components_count = 0;
3024 const CharT* end_of_share_name = nullptr;
3025 const auto* start = first;
3026 while (start != last) {
3027 const auto* pcend = std::find_if(start, last, detail::is_windows_slash<CharT>);
3028 // path components MUST be at least one character in length
3029 if (start == pcend)
3030 return nullptr;
3031 // path components MUST NOT contain a backslash (\‍) or a null
3032 if (std::find(start, pcend, '\0') != pcend)
3033 return nullptr;
3034
3035 ++path_components_count;
3036
3037 switch (path_components_count) {
3038 case 1:
3039 // Check the first UNC path component (hostname)
3040 switch (pcend - start) {
3041 case 1:
3042 // Do not allow "?" and "." hostnames, because "\\?\" means Win32 file
3043 // namespace and "\\.\" means Win32 device namespace
3044 if (start[0] == '?' || start[0] == '.')
3045 return nullptr;
3046 break;
3047 case 2:
3048 // Do not allow Windows drive letter, because it is not a valid hostname
3049 if (detail::is_windows_drive(start[0], start[1]))
3050 return nullptr;
3051 break;
3052 }
3053 // Accept UNC path with hostname, even if it does not contain share-name
3054 end_of_share_name = pcend;
3055 break;
3056 case 2:
3057 // Check the second UNC path component (share name).
3058 // Do not allow "." and ".." as share names, because they have
3059 // a special meaning and are removed by the URL parser.
3060 switch (pcend - start) {
3061 case 1:
3062 if (start[0] == '.')
3063 return nullptr;
3064 break;
3065 case 2:
3066 if (start[0] == '.' && start[1] == '.')
3067 return nullptr;
3068 break;
3069 }
3070 // A valid UNC path MUST contain two or more path components
3071 end_of_share_name = pcend;
3072 break;
3073 default:;
3074 }
3075 if (pcend == last) break;
3076 start = pcend + 1; // skip '\'
3077 }
3078 return end_of_share_name;
3079}

3088inline bool has_dot_dot_segment(const CharT* first, const CharT* last, IsSlash is_slash) {
3089 if (last - first >= 2) {

3095 (last - ptr == 2 || is_slash(ptr[2])))
3096 return true;
3097 // skip '.' and following char
3098 ptr += 2;
3099 if (ptr >= end)
3100 break;
3101 }
3102 }
3103 return false;
3104}

3211 detail::is_windows_slash(pointer[1])) {

3213 pointer += 2; // skip "?\" or ".\"

3215 (pointer[0] | 0x20) == 'u' &&
3216 (pointer[1] | 0x20) == 'n' &&

3218 detail::is_windows_slash(pointer[3])) {

3222 } else {

3225 }

3241 // make URL

3254inline std::string path_from_file_url(const url& file_url, file_path_format format = file_path_format::native) {
3255 if (!file_url.is_file_scheme())
3256 throw url_error(validation_errc::not_file_url, "Not a file URL");
3257
3258 // source
3259 const auto hostname = file_url.hostname();
3260 const bool is_host = !hostname.empty();
3261
3262 // target
3263 std::string path;
3264
3265 if (format == file_path_format::posix) {
3266 if (is_host)
3267 throw url_error(validation_errc::file_url_cannot_have_host, "POSIX path cannot have host");
3268 // percent decode pathname
3269 detail::append_percent_decoded(file_url.pathname(), path);
3270 } else {
3271 // format == file_path_format::windows
3272 if (is_host) {
3273 // UNC path cannot have "." hostname, because "\\.\" means Win32 device namespace
3274 if (hostname == ".")
3275 throw url_error(validation_errc::file_url_unsupported_host, "UNC path cannot have \".\" hostname");

3279 // Form an IPV6 address host-name by substituting hyphens for the colons and appending ".ipv6-literal.net"
3280 // https://learn.microsoft.com/en-us/openspecs/windows_protocols/ms-dtyp/62e862f4-2a51-452e-8eeb-dc4ff5ee33cc
3281 std::replace_copy(std::next(hostname.begin()), std::prev(hostname.end()),
3282 std::back_inserter(path), ':', '-');
3283 path.append(".ipv6-literal.net");
3284 } else {
3285 path.append(hostname);
3286 }
3287 }
3288
3289 // percent decode pathname and normalize slashes
3290 const auto start = static_cast<std::ptrdiff_t>(path.length());
3291 detail::append_percent_decoded(file_url.pathname(), path);
3292 std::replace(std::next(path.begin(), start), path.end(), '/', '\\');
3293
3294 if (is_host) {
3295 if (!detail::is_unc_path(path.data() + 2, path.data() + path.length()))
3296 throw url_error(validation_errc::file_url_invalid_unc, "Invalid UNC path");
3297 } else {

3299 path.erase(0, 1); // remove leading '\\'

3301 path.push_back('\\'); // "C:" -> "C:\"
3302 } else {
3303 // https://datatracker.ietf.org/doc/html/rfc8089#appendix-E.3.2
3304 // Maybe a UNC path. Possible variants:
3305 // 1) file://///host/path -> \\\host\path
3306 // 2) file:////host/path -> \\host\path
3307 const auto count_leading_slashes = std::find_if(
3308 path.data(),
3309 path.data() + std::min(static_cast<std::size_t>(4), path.length()),
3310 [](char c) { return c != '\\'; }) - path.data();

3312 path.erase(0, 1); // remove leading '\\'
3313 else if (count_leading_slashes != 2)
3314 throw url_error(validation_errc::file_url_not_windows_path, "Not a Windows path");
3315 if (!detail::is_unc_path(path.data() + 2, path.data() + path.length()))
3316 throw url_error(validation_errc::file_url_invalid_unc, "Invalid UNC path");
3317 }
3318 }
3319 }

3321 // Check for null characters
3322 if (util::contains_null(path.begin(), path.end()))
3323 throw url_error(validation_errc::null_character, "Path contains null character");
