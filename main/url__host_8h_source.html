148// The host parser
149// https://url.spec.whatwg.org/#concept-host-parser
150
151template <typename CharT>
152inline validation_errc host_parser::parse_host(const CharT* first, const CharT* last, bool is_opaque, host_output& dest) {
153 using UCharT = typename std::make_unsigned<CharT>::type;
154
155 // 1. Non-"file" special URL's cannot have an empty host.
156 // 2. For "file" URL's empty host is set in the file_host_state 1.2
157 // https://url.spec.whatwg.org/#file-host-state
158 // 3. Non-special URLs can have empty host and it will be set here.
159 if (first == last) {
160 // https://github.com/whatwg/url/issues/79
161 // https://github.com/whatwg/url/pull/189
162 // set empty host
163 dest.hostStart();
164 dest.hostDone(HostType::Empty);
165 return is_opaque ? validation_errc::ok : validation_errc::host_missing;
166 }
167 assert(first < last);
168
169 if (*first == '[') {
170 if (*(last - 1) == ']') {
171 return parse_ipv6(first + 1, last - 1, dest);
172 }
173 // 1.1. If input does not end with U+005D (]), IPv6-unclosed
174 // validation error, return failure.
175 return validation_errc::ipv6_unclosed;
176 }
177
178 if (is_opaque)
179 return parse_opaque_host(first, last, dest);
180
181 // Is ASCII domain?
182 const auto ptr = std::find_if_not(first, last, detail::is_ascii_domain_char<CharT>);
183 if (ptr == last) {
184 if (!util::has_xn_label(first, last)) {
185 // Fast path for ASCII domain
186
187 // If asciiDomain ends in a number, return the result of IPv4 parsing asciiDomain
188 if (hostname_ends_in_a_number(first, last))
189 return parse_ipv4(first, last, dest);
190
191 // Return asciiDomain lower cased
192 std::string& str_host = dest.hostStart();
193 util::append_ascii_lowercase(str_host, first, last);
194 dest.hostDone(HostType::Domain);
195 return validation_errc::ok;
196 }
197 } else if (static_cast<UCharT>(*ptr) < 0x80 && *ptr != '%') {
198 // NFC normalizes U+003C (<), U+003D (=), U+003E (>) characters if they precede
199 // U+0338. Therefore, no errors are reported here for forbidden < and > characters
200 // if there is a possibility to normalize them.
201 if (!(*ptr >= 0x3C && *ptr <= 0x3E && ptr + 1 < last && static_cast<UCharT>(ptr[1]) >= 0x80))
202 // 7. If asciiDomain contains a forbidden domain code point, domain-invalid-code-point
203 // validation error, return failure.
204 return validation_errc::domain_invalid_code_point;
205 }
206
207 // Input for domain_to_ascii
208 simple_buffer<char16_t> buff_uc;
209
210 // copy ASCII chars
211 for (auto it = first; it != ptr; ++it) {
212 const auto uch = static_cast<UCharT>(*it);
213 buff_uc.push_back(static_cast<char16_t>(uch));
214 }
215
216 // Let buff_uc be the result of running UTF-8 decode (to UTF-16) without BOM
217 // on the percent decoding of UTF-8 encode on input
218 for (auto it = ptr; it != last;) {
219 const auto uch = static_cast<UCharT>(*it++);
220 if (uch < 0x80) {
221 if (uch != '%') {
222 buff_uc.push_back(static_cast<char16_t>(uch));
223 continue;
224 }
225 // uch == '%'
226 unsigned char uc8; // NOLINT(cppcoreguidelines-init-variables)
227 if (detail::decode_hex_to_byte(it, last, uc8)) {
228 if (uc8 < 0x80) {
229 buff_uc.push_back(static_cast<char16_t>(uc8));
230 continue;
231 }
232 // percent encoded utf-8 sequence
233 // TODO: gal po vieną code_point, tuomet užtektų utf-8 buferio vienam simboliui
234 simple_buffer<char> buff_utf8;
235 buff_utf8.push_back(static_cast<char>(uc8));
236 while (it != last && *it == '%') {
237 ++it; // skip '%'
238 if (!detail::decode_hex_to_byte(it, last, uc8))
239 uc8 = '%';
240 buff_utf8.push_back(static_cast<char>(uc8));
241 }
242 url_utf::convert_utf8_to_utf16(buff_utf8.data(), buff_utf8.data() + buff_utf8.size(), buff_uc);
243 //buff_utf8.clear();
244 continue;
245 }
246 // detected an invalid percent-encoding sequence
247 buff_uc.push_back('%');
248 } else { // uch >= 0x80
249 --it;
250 url_utf::append_utf16(url_utf::read_utf_char(it, last).value, buff_uc);
251 }
252 }
253
254
255 // domain to ASCII
256 simple_buffer<char16_t> buff_ascii;
257
258 const auto res = domain_to_ascii(buff_uc.data(), buff_uc.size(), buff_ascii);
259 if (res != validation_errc::ok)
260 return res;
261 if (detail::contains_forbidden_domain_char(buff_ascii.data(), buff_ascii.data() + buff_ascii.size())) {
262 // 7. If asciiDomain contains a forbidden domain code point, domain-invalid-code-point
263 // validation error, return failure.
264 return validation_errc::domain_invalid_code_point;
265 }
266
267 // If asciiDomain ends in a number, return the result of IPv4 parsing asciiDomain
268 if (hostname_ends_in_a_number(buff_ascii.begin(), buff_ascii.end()))
269 return parse_ipv4(buff_ascii.begin(), buff_ascii.end(), dest);
270
271 // Return asciiDomain
272 std::string& str_host = dest.hostStart();
273 util::append(str_host, buff_ascii);
274 dest.hostDone(HostType::Domain);
275 return validation_errc::ok;
276}

278// The opaque-host parser
279// https://url.spec.whatwg.org/#concept-opaque-host-parser

281template <typename CharT>
282inline validation_errc host_parser::parse_opaque_host(const CharT* first, const CharT* last, host_output& dest) {
283 // 1. If input contains a forbidden host code point, host-invalid-code-point
284 // validation error, return failure.
285 if (detail::contains_forbidden_host_char(first, last))
286 return validation_errc::host_invalid_code_point;
287
288 // TODO-WARN:
289 // 2. If input contains a code point that is not a URL code point and not U+0025 (%),
290 // invalid-URL-unit validation error.
291 // 3. If input contains a U+0025 (%) and the two code points following it are not ASCII hex digits,
292 // invalid-URL-unit validation error.
293
294 std::string& str_host = dest.hostStart();
295
296 //TODO: UTF-8 percent encode it using the C0 control percent-encode set
297 //detail::append_utf8_percent_encoded(first, last, detail::CHAR_C0_CTRL, str_host);
298 using UCharT = typename std::make_unsigned<CharT>::type;
299
300 const CharT* pointer = first;
301 while (pointer < last) {
302 // UTF-8 percent encode c using the C0 control percent-encode set (U+0000 ... U+001F and >U+007E)
303 const auto uch = static_cast<UCharT>(*pointer);
304 if (uch >= 0x7f) {
305 // invalid utf-8/16/32 sequences will be replaced with 0xfffd
306 detail::append_utf8_percent_encoded_char(pointer, last, str_host);
307 } else {
308 // Just append the 7-bit character, percent encoding C0 control chars
309 const auto uc = static_cast<unsigned char>(uch);
310 if (uc <= 0x1f)
311 detail::append_percent_encoded_byte(uc, str_host);
312 else
313 str_host.push_back(uc);
314 ++pointer;
315 }
316 }
317
318 dest.hostDone(str_host.empty() ? HostType::Empty : HostType::Opaque);
319 return validation_errc::ok;
320}

322template <typename CharT>
323inline validation_errc host_parser::parse_ipv4(const CharT* first, const CharT* last, host_output& dest) {
324 uint32_t ipv4; // NOLINT(cppcoreguidelines-init-variables)
325
326 const auto res = ipv4_parse(first, last, ipv4);
327 if (res == validation_errc::ok) {
328 std::string& str_ipv4 = dest.hostStart();
329 ipv4_serialize(ipv4, str_ipv4);
330 dest.hostDone(HostType::IPv4);
331 }
332 return res;
333}

335template <typename CharT>
336inline validation_errc host_parser::parse_ipv6(const CharT* first, const CharT* last, host_output& dest) {
337 uint16_t ipv6addr[8];
338
339 const auto res = ipv6_parse(first, last, ipv6addr);
340 if (res == validation_errc::ok) {
341 std::string& str_ipv6 = dest.hostStart();
342 str_ipv6.push_back('[');
343 ipv6_serialize(ipv6addr, str_ipv6);
344 str_ipv6.push_back(']');
345 dest.hostDone(HostType::IPv6);
346 }
347 return res;
348}
