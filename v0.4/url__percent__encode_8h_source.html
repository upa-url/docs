394// ----------------------------------------------------------------------------
395// Hex digit conversion tables and functions
396
397// Maps the hex numerical values 0x0 to 0xf to the corresponding ASCII digit
398// that will be used to represent it.
399extern const char kHexCharLookup[0x10];
400
401// This lookup table allows fast conversion between ASCII hex letters and their
402// corresponding numerical value. The 8-bit range is divided up into 8
403// regions of 0x20 characters each. Each of the three character types (numbers,
404// uppercase, lowercase) falls into different regions of this range. The table
405// contains the amount to subtract from characters in that range to get at
406// the corresponding numerical value.
407//
408// See hex_char_to_num for the lookup.
409extern const char kCharToHexLookup[8];
410
411// Assumes the input is a valid hex digit! Call is_hex_char before using this.
412inline unsigned char hex_char_to_num(unsigned char c) noexcept {
413 return c - kCharToHexLookup[c / 0x20];
414}
415
416// ----------------------------------------------------------------------------
417// Percent decode
418
419// Given a character after '%' at |*first| in the string, this will decode
420// the escaped value and put it into |*unescaped_value| on success (returns
421// true). On failure, this will return false, and will not write into
422// |*unescaped_value|.
423//
424// |*first| will be updated to point after the last character of the escape
425// sequence. On failure, |*first| will be unchanged.
426
427template <typename CharT>
428inline bool decode_hex_to_byte(const CharT*& first, const CharT* last, unsigned char& unescaped_value) {
429 if (last - first < 2 ||
430 !is_hex_char(first[0]) || !is_hex_char(first[1])) {
431 // not enough or invalid hex digits
432 return false;
433 }
434
435 // Valid escape sequence.
436 const auto uc1 = static_cast<unsigned char>(first[0]);
437 const auto uc2 = static_cast<unsigned char>(first[1]);
438 unescaped_value = (hex_char_to_num(uc1) << 4) + hex_char_to_num(uc2);
439 first += 2;
440 return true;
441}

443// ----------------------------------------------------------------------------
444// Percent encode
445
446// Percent-encodes byte and appends to string
447// See: https://url.spec.whatwg.org/#percent-encode
448
449inline void append_percent_encoded_byte(unsigned char uc, std::string& output) {
450 output.push_back('%');
451 output.push_back(kHexCharLookup[uc >> 4]);
452 output.push_back(kHexCharLookup[uc & 0xf]);
453}

455// Reads one character from string (first, last), converts to UTF-8, then
456// percent-encodes, and appends to `output`. Replaces invalid UTF-8, UTF-16 or UTF-32
457// sequences in input with Unicode replacement characters (U+FFFD) if present.
458
459template <typename CharT>
460inline bool append_utf8_percent_encoded_char(const CharT*& first, const CharT* last, std::string& output) {
461 // url_util::read_utf_char(..) will handle invalid characters for us and give
462 // us the kUnicodeReplacementCharacter, so we don't have to do special
463 // checking after failure, just pass through the failure to the caller.
464 const auto cp_res = url_utf::read_utf_char(first, last);
465 // convert cp_res.value code point to UTF-8, then percent encode and append to `output`
466 url_utf::append_utf8<std::string, append_percent_encoded_byte>(cp_res.value, output);
467 return cp_res.result;
468}
469
470// Converts input string (first, last) to UTF-8, then percent encodes bytes not
471// in `cpset`, and appends to `output`. Replaces invalid UTF-8, UTF-16 or UTF-32
472// sequences in input with Unicode replacement characters (U+FFFD) if present.
473
474template<typename CharT>
475inline void append_utf8_percent_encoded(const CharT* first, const CharT* last, const code_point_set& cpset, std::string& output) {
476 using UCharT = typename std::make_unsigned<CharT>::type;
477
478 for (auto it = first; it < last; ) {
479 const auto uch = static_cast<UCharT>(*it);
480 if (uch >= 0x80) {
481 // invalid utf-8/16/32 sequences will be replaced with kUnicodeReplacementCharacter
482 append_utf8_percent_encoded_char(it, last, output);
483 } else {
484 // Just append the 7-bit character, possibly percent encoding it
485 const auto uc = static_cast<unsigned char>(uch);
486 if (is_char_in_set(uc, cpset)) {
487 output.push_back(uc);
488 } else {
489 // other characters are percent encoded
490 append_percent_encoded_byte(uc, output);
491 }
492 ++it;
493 }
494 }
495}

506template <class StrT, enable_if_str_arg_t<StrT> = 0>
507inline void append_percent_decoded(StrT&& str, std::string& output) {
508 const auto inp = make_str_arg(std::forward<StrT>(str));
509 const auto* first = inp.begin();
510 const auto* last = inp.end();
511
512 for (auto it = first; it != last;) {
513 const auto uch = util::to_unsigned(*it); ++it;
514 if (uch < 0x80) {
515 if (uch != '%') {
516 output.push_back(static_cast<char>(uch));
517 continue;
518 }
519 // uch == '%'
520 unsigned char uc8; // NOLINT(cppcoreguidelines-init-variables)
521 if (decode_hex_to_byte(it, last, uc8)) {
522 if (uc8 < 0x80) {
523 output.push_back(static_cast<char>(uc8));
524 continue;
525 }
526 // percent encoded utf-8 sequence
527 std::string buff_utf8;
528 buff_utf8.push_back(static_cast<char>(uc8));
529 while (it != last && *it == '%') {
530 ++it; // skip '%'
531 if (!decode_hex_to_byte(it, last, uc8))
532 uc8 = '%';
533 buff_utf8.push_back(static_cast<char>(uc8));
534 }
535 url_utf::check_fix_utf8(buff_utf8);
536 output += buff_utf8;
537 continue;
538 }
539 // detected invalid percent encoding
540 output.push_back('%');
541 } else { // uch >= 0x80
542 --it;
543 url_utf::read_char_append_utf8(it, last, output);
544 }
545 }
